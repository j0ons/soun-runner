# PDF generation (optional — HTML report works without it).
# Primary engine: Playwright (headless Chromium). Fully portable — ships its
# own browser, needs NO system libraries, so it works on client Windows
# machines where WeasyPrint fails with "cannot load library libgobject-2.0-0".
# After installing, run ONCE to download the browser:  playwright install chromium
playwright>=1.40.0

# Fallback engine: WeasyPrint. Needs native GTK (pango/cairo/gobject) present
# on the machine. Used only if Playwright/Chromium is unavailable.
weasyprint>=60.0

# NOTE: Network auto-detection, ISP/ASN lookup, traceroute topology,
# CVE lookup (NIST NVD), and SSL/TLS analysis all use the Python standard
# library only (socket, ssl, subprocess, urllib) — no extra packages needed.
# Nmap must be installed separately on the host (winget/brew install nmap).

# Remote fix execution ("Run the fix") — OPTIONAL. Local fixes run with the
# stdlib (PowerShell/bash subprocess); these are only needed to push a fix to a
# REMOTE host. The app imports them lazily and degrades gracefully if absent
# (the UI shows "remote execution unavailable — install paramiko / pywinrm").
#   paramiko  -> SSH to Linux targets
#   pywinrm   -> WinRM to Windows targets
paramiko>=3.0.0
pywinrm>=0.4.3
